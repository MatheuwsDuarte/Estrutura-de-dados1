    int auxv1=0;
    int auxv2=0;
    
    while (aux1 != NULL && aux1->valor != M)

    }

    while (aux2 != NULL && aux2->valor != N)

        aux2 = aux2->prox;  
        
    }

    auxv1 = aux1-> valor;

      //CASO aux comeÃ§o

          aux2 = aux1;

        {
          aux1 = aux1 -> ant;

          aux1 -> valor = auxv1;

        }

              aux2 = aux2-> prox;

        aux2->ant->prox -> valor = auxv1;

    aux = aux1->valor;

}
